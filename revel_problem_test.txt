double dist(double a,double b)
{   
    // vars
    double y, x, result;
    // subtracts the diff
    y = a.y - b.y;
    x = a.x - b.x;
    //squares the differences
    y = pow(y,2);
    x = pow(x,2);
    // adding of the squares
    result = sqrt(x + y);
    return result;
}
